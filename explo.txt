cat << "EOF" > test2.mx3
Aex = 16e-12
Msat=16e5
Ku1= 2e6
Alpha = 1e-3
dind=3e-3

SetGridsize(230, 90, 1)       
SetCellsize(1e-9, 1e-9, 1e-9)  

a:=cuboid(300e-9, 10e-9,1e-9)
b:=cylinder(40e-9,1e-9).transl(0, 25e-9, 0)
rect1 := rect(4e-9,40e-9).transl(-110e-9,0,0)
write1 := rect(4e-9,40e-9).transl(-25e-9,0,0)
write2 := rect(4e-9,40e-9).transl(-35e-9,0,0)
write3 := rect(4e-9,40e-9).transl(-45e-9,0,0)
write4 := rect(4e-9,40e-9).transl(-55e-9,0,0)

read1 := rect(4e-9,40e-9).transl(25e-9,0,0)
read2 := rect(4e-9,40e-9).transl(35e-9,0,0)
read3 := rect(4e-9,40e-9).transl(45e-9,0,0)
read4 := rect(4e-9,40e-9).transl(55e-9,0,0)
defregion(2,write1)
defregion(3,write2)
defregion(4,write3)
defregion(5,write4)
defregion(6,read1)
defregion(7,read2)
defregion(8,read3)
defregion(9,read4)
defregion(1,rect1)
setgeom(a.add(b))
saveas(geom, "logicAdd")
h  := 10e-3
fcut := 100e9
Bx := h * (sinc(2*pi*fcut*t))
B_ext = vector(0,0,0)
B_ext.SetRegion(1, vector(Bx,0,0))

m=uniform(0,0,1)
relax()
save(m)
save(regions)

for i:=2; i<10; i++{
   tableadd(m.region(i))
}

tableautosave(5e-12)
run(20e-9)
EOF
